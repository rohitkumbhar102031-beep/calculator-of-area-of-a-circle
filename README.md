# calculator-of-area-of-a-circle
calculating the area of a circle when the radius is given<br> 
input:-<br>
#include <stdio.h>
/* calculating the are of a circle from the given formula:
area of a circle = pi*r*r */
int main(){
    float pi ('3.14');
    int radius;
    printf ("enter radius");
    scanf ("%f", &radius);
    printf("area is: %f ", pi*radius*radius);
    return 0 ;
}
